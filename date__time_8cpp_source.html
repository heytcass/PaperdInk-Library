22
23 pdink_now.month_num = timeinfo.tm_mon + 1;
24 // Gives offset of first day of the month with respect to Monday
25 // https://www.tondering.dk/claus/cal/chrweek.php#calcdow
26 // 0=Sunday, 1=Monday ... 6=Saturday
27 uint16_t a = (14 - pdink_now.month_num) / 12;
28 uint16_t y = pdink_now.year - a;
29 uint16_t m = pdink_now.month_num + (12 * a) - 2;
30
31 pdink_now.day_offset = (1 + y + (y / 4) - (y / 100) + (y / 400) + ((31 * m) / 12)) % 7;
32 // Change the offset based on user preference
33 pdink_now.day_offset = (pdink_now.day_offset + week_start_offset) % 7;
34
35 // Convert to 12 hour
36 if(pdink_now.mil_hour > 12){
37 pdink_now.hour = pdink_now.mil_hour - 12;
38 }else{
39 pdink_now.hour = pdink_now.mil_hour;
40 }
41
42 Serial.printf("Time is %d %d:%d:%d on %s on the %d/%d/%d . It is the month of %s. day_offset: %d\r\n", pdink_now.mil_hour, pdink_now.hour, pdink_now.min, pdink_now.sec, pdink_now.wday, pdink_now.mday, pdink_now.month_num, pdink_now.year, pdink_now.month, pdink_now.day_offset);

44}
